64 {

67  int i; // pour la boucle for

73  // On mélange ce tableau
74  int *t = (int *)calloc(47, sizeof(int)); // ce tableau sert à savoir si l'indice généré a déjà été généré (t[i]=1 si i déjà sorti, 0 sinon)
75  Carte *pioche_melange = (Carte *)malloc(47 * sizeof(Carte));
76  int c = 0; // sert à compter le nombre de cartes rentrées dans pioche_melange
77  int n = rand() % 47;
78  while (c != 47)

81  { // tant que l'indice a déjà été traité
82  n = rand() % 47; // génère un nombre entier aléatoire entre 0 et 46
83  }
84  pioche_melange[c] = pioche[n]; // on remplit pioche_melange grâce à pioche
85  t[n] = 1; // on indique que n a été traité
86  c += 1;
87  }
